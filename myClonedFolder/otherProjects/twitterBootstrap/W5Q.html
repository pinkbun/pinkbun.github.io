            function getFooBar() {
                for (var i = 1; i <= 100; i++) {
                    var isDividibleByThree = i % 3 === 0;
                    var isDivisibleByFive = i % 5 === 0;
                    var print = "";

                    if (isDividibleByThree && isDivisibleByFive) {
                        console.log('FooBar');
                        print = print + "FooBar" + "
";

                    } else if (isDividibleByThree) {
                        console.log('Foo');
                        print = print + "Foo" + "
";

                    } else if (isDivisibleByFive) {
                        console.log('Bar');
                        print = print + 'Bar' + "
";

                    } else {
                        console.log(i);
                        print = print + i + "
";

                    }
                }
                $("‪#Q2result‬").html(print);
            }
