    function countBs(bString) {
        var count = 0;
        for (var i = 0; i < bString.length; i++) {
            if (bString.charAt(i) === 'B') {
                count = (count + 1);
            }
        }
        return count;
    }

          function countChar(cString, charMatch) {
        var count = 0;
        for (var i = 0; i < cString.length; i++) {
          if (cString.charAt(i) == charMatch) {
            count = (count + 1);
          }
        }
        return count;
      }

function range(end, start) {
    var arr = [];
    //arr.push(5);
    //arr.push(1);
    for (end; end >= start; end--) {
        arr.push(end);

    }

    return arr;
}
